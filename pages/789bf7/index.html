天地图的简单使用

from flask import Flask, request
from flask_cors import CORS
import pymysql
import json

app = Flask(__name__)
cors = CORS(app)

@app.route('/get_gps', methods=['GET'])
def get_gps():
    db = pymysql.connect(host='xxx', user='xxx', password='xx', db='position', charset='utf8')
    cursor = db.cursor()
    sql = "SELECT * FROM gps ORDER BY id DESC LIMIT 1"
    cursor.execute(sql)
    gps = cursor.fetchone()
    print(gps)
    result = {}
    result['lat'] = str(gps[1])
    result['lng'] = str(gps[2])
    return json.dumps(result)

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=8080)
